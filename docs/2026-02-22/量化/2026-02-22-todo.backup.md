> **详细规划文档**: [PHASE4_DETAILED_PLAN.md](./PHASE4_DETAILED_PLAN.md)

### 12.1 核心哲学

**三不做**:
- ❌ 不预测明天会怎样
- ❌ 不赌未来走势
- ❌ 不坚持自己的判断

**三确认**:
- ✅ 只确认现在在发生什么
- ✅ 只响应市场给出的信号
- ✅ 只在市场给出答案的那一刻行动

- 验证成功: +0.1
- 验证失败: -0.3
- 共振达成: +0.2
- 发现诱多: -0.5 (立即退出)
- 验证点失败: 立即离场 (置信度 > 0.8)
```

# 买入后立即创建验证点

#### 14.5.3 双向验证的核心

**不仅仅是预测对错，还要确认原因一致**：

```python
# 传统验证
预测: 涨2% → 实际: 涨2% → 结果: ✓ 正确

# 双向验证
预测: 涨2% (理由: 突破确认) → 实际: 涨2%
                                              ↓
                                    检查实际原因
                                              ↓
                           实际原因: 重大利好消息 → 结果: ⚠️ 原因不一致
```

**原因一致性检查** (`grader.py`):
- 提取预测中的预期原因
- 找出实际走势的真正原因
- 比较两者是否一致

#### 19.3.1 持仓心理模块 - 完整实现

```python
"""
Phase 4 Week 3: 持仓心理模块

让系统对抗人性弱点：恐高、贪婪、犹豫、侥幸、后悔
"""

from dataclasses import dataclass
from typing import List, Dict, Any, Optional
from datetime import datetime, timedelta


@dataclass
class PsychologicalIssue:
    """心理问题"""
    issue: str
    severity: float  # 0-1
    description: str
    suggested_action: str


@dataclass
class PsychologyReport:
    """心理分析报告"""
    has_issues: bool
    issues: List[PsychologicalIssue]
    hold_confidence: float  # 0-1
    suggested_action: str  # hold, sell, partial_sell, watch
    reasoning: str


class HoldingPsychology:
    """持仓心理模块 - 让系统对抗人性弱点"""

    def __init__(self):
        # 心理阈值配置
        self.thresholds = {
            'fear_of_height_profit': 0.20,  # 涨20%开始恐高
            'greed_profit': 0.30,               # 涨30%开始贪婪
            'hesitation_stop_distance': 0.01,  # 距止损位1%开始犹豫
            'wishful_thinking_loss': -0.02,    # 亏2%开始侥幸
            'regret_sensitivity': 0.10,          # 卖早10%开始后悔
        }

    def analyze(
        self,
        position: Dict,
        current_price: float,
        wave_assessment: Dict,
        target_prediction: Optional[Dict] = None,
    ) -> PsychologyReport:
        """
        分析当前持仓心理状态

        Args:
            position: 持仓信息 {entry_price, entry_time, quantity, ...}
            current_price: 当前价格
            wave_assessment: 主升浪评估结果
            target_prediction: 目标价预测结果

        Returns:
            PsychologyReport: 心理分析报告
        """
        profit = (current_price - position['entry_price']) / position['entry_price']
        profit_pct = profit * 100

        issues = []
        hold_confidence = 0.5  # 基础信心 50%

        # 1. 恐高症检查
        if self._check_fear_of_height(profit, wave_assessment):
            issues.append(PsychologicalIssue(
                issue='恐高症',
                severity=self._calc_fear_severity(profit, wave_assessment),
                description=f'已盈利{profit_pct:.1f}，想落袋为安',
                suggested_action='主升浪强度高，建议继续持有'
            ))
            # 恐高打击信心
            hold_confidence -= 0.15

        # 2. 贪婪症检查
        if self._check_greed(profit, wave_assessment):
            issues.append(PsychologicalIssue(
                issue='贪婪症',
                severity=self._calc_greed_severity(profit, wave_assessment),
                description=f'已盈利{profit_pct:.1f}，想赚更多',
                suggested_action='主升浪强度已下降，建议分批止盈'
            ))
            # 贪婪也打击信心（过度贪婪可能）
            hold_confidence -= 0.20

        # 3. 犹豫症检查
        stop_loss = position.get('stop_loss')
        if stop_loss and abs(current_price - stop_loss) / stop_loss < 0.01:
            issues.append(PsychologicalIssue(
                issue='犹豫症',
                severity=0.6,
                description=f'价格${current_price:.2f}接近止损位${stop_loss:.2f}',
                suggested_action='止损是保护机制，到了就执行'
            ))
            hold_confidence -= 0.10

        # 4. 侥幸症检查
        if profit < 0 and self._check_wishful_thinking(profit, wave_assessment):
            issues.append(PsychologicalIssue(
                issue='侥幸症',
                severity=0.7,
                description=f'亏损{profit_pct:.1f}，觉得会反弹',
                suggested_action='趋势已弱，建议止损离场，保住本金'
            ))
            hold_confidence -= 0.25

        # 5. 后悔症检查
        if position.get('sold_too_early_penalty'):
            issues.append(PsychologicalIssue(
                issue='后悔症',
                severity=0.5,
                description=f'上次卖早了{position["sold_too_early_penalty"]:.1%}，想多拿',
                suggested_action='每次交易独立看待，不要被过去影响'
            ))
            hold_confidence -= 0.10

        # 计算最终信心
        # 基础信心：主升浪强度
        base_confidence = wave_assessment.get('total_score', 0.5)

        # 盈利影响：适度盈利增加信心，过度盈利或亏损减少信心
        if 0.05 < profit < 0.25:
            hold_confidence += 0.15  # 健康盈利区间
        elif profit >= 0.25:
            hold_confidence -= 0.10   # 过度盈利恐高
        elif profit < -0.02:
            hold_confidence -= 0.30   # 亏损打击信心

        # 心理问题影响
        issue_penalty = sum(issue.severity for issue in issues)
        hold_confidence -= issue_penalty

        # 确保信心在合理范围
        hold_confidence = max(0.0, min(1.0, base_confidence + hold_confidence))

        # 生成行动建议
        suggested_action = self._generate_suggested_action(
            profit, wave_assessment, target_prediction, hold_confidence
        )

        return PsychologyReport(
            has_issues=len(issues) > 0,
            issues=issues,
            hold_confidence=hold_confidence,
            suggested_action=suggested_action,
            reasoning=self._generate_reasoning(issues, wave_assessment, profit)
        )

    def _check_fear_of_height(self, profit: float, wave: Dict) -> bool:
        """检查恐高症"""
        return (
            profit > self.thresholds['fear_of_height_profit']
            and wave.get('total_score', 0) > 0.6  # 主升浪强势
        )

    def _check_greed(self, profit: float, wave: Dict) -> bool:
        """检查贪婪症"""
        return (
            profit > self.thresholds['greed_profit']
            and wave.get('total_score', 0) < 0.4  # 主升浪转弱
        )

    def _check_wishful_thinking(self, profit: float, wave: Dict) -> bool:
        """检查侥幸症"""
        return (
            profit < self.thresholds['wishful_thinking_loss']
            and wave.get('total_score', 0) < 0.3  # 趋势已弱
        )

    def _calc_fear_severity(self, profit: float, wave: Dict) -> float:
        """计算恐高严重程度"""
        base_severity = min(profit / 0.50, 1.0)  # 50%盈利 = 1.0严重
        wave_discount = 1 - wave.get('total_score', 0.5)  # 强势降低严重度
        return base_severity * wave_discount

    def _calc_greed_severity(self, profit: float, wave: Dict) -> float:
        """计算贪婪严重程度"""
        return min((profit - 0.30) / 0.70, 1.0)  # 100%盈利 = 1.0

    def _generate_suggested_action(
        self,
        profit: float,
        wave: Dict,
        target: Optional[Dict],
        confidence: float
    ) -> str:
        """生成行动建议"""
        if wave.get('total_score', 0) > 0.7 and profit > 0:
            return 'strong_hold'  # 强势主升浪，坚定持有
        elif wave.get('total_score', 0) > 0.5 and profit > 0.05:
            return 'hold'  # 正常持有
        elif confidence > 0.6:
            return 'hold_with_caution'  # 谨慎持有
        elif confidence < 0.4:
            return 'consider_sell'  # 考虑卖出
        elif profit < -0.03:
            return 'sell'  # 亏损离场
        else:
            return 'watch'  # 观察

    def _generate_reasoning(
        self,
        issues: List[PsychologicalIssue],
        wave: Dict,
        profit: float
    ) -> str:
        """生成建议理由"""
        if not issues:
            return f"盈利{profit:.1%}，主升浪强度{wave['total_score']:.1%}，正常持有"

        reasons = [f"{i.issue}: {i.suggested_action}" for i in issues]
        return "; ".join(reasons)

                wave,

        if hold_time > self.style.get('hold_time_max', 240):
            return True, '持仓超时'

        if volume_ratio > 3:

        signals = []

        ma5 = data['close'].rolling(5).mean()
        ma20 = data['close'].rolling(20).mean()

                'name': 'golden_cross',
                'value': 1.0,
                'weight': 0.3

class StyleAdapter:
    """风格适配器 - 让系统适配不同交易风格"""

    def __init__(self, style_def):
        self.style_def = style_def
        self.current_style = None
        self.adapters = {}

    def set_style(self, style_name):
        """设置当前交易风格"""
        style = self.style_def.get_style(style_name)
        if not style:
            raise ValueError(f"未知风格: {style_name}")

        self.current_style = style
        print(f"切换到 {style_name} 模式")
        print(f"  特征: {style['description']}")
        print(f"  持仓时间: {style['hold_time_min']}-{style['hold_time_max']} {style['timeframe']}")
        print(f"  目标盈利: {style['target_profit']:.1%}")
        print(f"  止损: {style['stop_loss']:.1%}")

        return style

    def adapt_system(self, system):
        """调整系统参数以适应当前风格"""
        if not self.current_style:
            return system

        # 1. 调整信号层
        system.perception.signal_weights = self.current_style['signals']

        # 2. 调整确认层
        system.confirmation.confirmation_time = self.current_style['confirmation_time']
        system.confirmation.volume_requirement = self.current_style['volume_requirement']

        # 3. 调整资金管理
        system.money_mgmt.max_single_position = self.current_style['max_position']
        system.money_mgmt.target_profit = self.current_style['target_profit']
        system.money_mgmt.stop_loss = self.current_style['stop_loss']

        # 4. 调整交易时段
        system.trading_hours = self.current_style['trading_hours']

        # 5. 调整时间粒度
        system.timeframe = self.current_style['timeframe']

        return system

    def get_strategy_template(self):
        """获取当前风格的策略模板"""
        if not self.current_style:
            return None

        template_lib = StrategyTemplateLibrary(self.style_def)
        return template_lib.get_template(self.current_style['name'])

    def create_style_adapter(self, style_name, system_components):
        """为特定风格创建适配器"""

        style = self.style_def.get_style(style_name)
        if not style:
            return None

        adapter = {
            'style': style,
            'components': {},
            'active': False
        }

        # 为每个组件创建风格化版本
        for comp_name, component in system_components.items():
            adapted = self._adapt_component(component, style)
            adapter['components'][comp_name] = adapted

        self.adapters[style_name] = adapter
        return adapter

    def _adapt_component(self, component, style):
        """调整单个组件"""
        if hasattr(component, 'adapt_to_style'):
            return component.adapt_to_style(style)
        return component

    def switch_adapter(self, style_name):
        """切换到指定风格的适配器"""
        if style_name not in self.adapters:
            return False

        # 关闭当前
        if self.current_style:
            current_name = self.current_style['name']
            if current_name in self.adapters:
                self.adapters[current_name]['active'] = False

        # 开启新
        self.adapters[style_name]['active'] = True
        self.current_style = self.adapters[style_name]['style']

        print(f"已切换到 {style_name} 适配器")
        return True

    def get_active_components(self):
        """获取当前激活的组件"""
        if not self.current_style:
            return {}

        style_name = self.current_style['name']
        if style_name in self.adapters:
            return self.adapters[style_name]['components']

        return {}

class MultiStyleBacktester:
    """多风格回测引擎 - 同时回测多种风格并对比表现"""

    def __init__(self, style_def, data_provider):
        self.style_def = style_def
        self.data_provider = data_provider

        self.results = {}
        self.comparisons = {}

    def backtest_style(self, style_name, start_date, end_date, initial_capital=100000):
        """回测单个风格"""

        print(f"\n===== 回测风格: {style_name} =====")

        style = self.style_def.get_style(style_name)
        if not style:
            print(f"未知风格: {style_name}")
            return None

        # 创建策略
        template_lib = StrategyTemplateLibrary(self.style_def)
        strategy = template_lib.get_template(style_name)

        # 获取数据
        data = self.data_provider.get_data(
            start_date, end_date,
            timeframe=style['timeframe']
        )

        # 运行回测
        capital = initial_capital
        positions = []
        trades = []

        for date, day_data in data.items():
            # 生成信号
            signals = strategy.generate_signals(day_data)

            # 决策
            for signal in signals:
                if strategy.should_buy([signal], {'current_time': date}):
                    # 买入
                    buy_price = day_data['close']
                    quantity = int(capital * style['max_position'] / buy_price)

                    position = {
                        'entry_date': date,
                        'entry_price': buy_price,
                        'quantity': quantity,
                        'signal': signal
                    }
                    positions.append(position)
                    capital -= buy_price * quantity

                elif positions:
                    # 检查卖出
                    for pos in positions[:]:
                        should_sell, reason = strategy.should_sell(
                            pos, day_data['close'],
                            {'current_time': date}
                        )
                        if should_sell:
                            # 卖出
                            sell_price = day_data['close']
                            profit = (sell_price - pos['entry_price']) / pos['entry_price']

                            trade = {
                                'entry_date': pos['entry_date'],
                                'exit_date': date,
                                'entry_price': pos['entry_price'],
                                'exit_price': sell_price,
                                'profit': profit,
                                'reason': reason,
                                'signal': pos['signal']
                            }
                            trades.append(trade)

                            capital += sell_price * pos['quantity']
                            positions.remove(pos)

        # 计算绩效
        result = self._calculate_performance(trades, capital, initial_capital)
        result['style'] = style_name
        result['trades'] = trades

        self.results[style_name] = result

        self._print_result(result)

        return result

    def backtest_multiple_styles(self, style_names, start_date, end_date):
        """回测多种风格"""

        print(f"\n{'='*60}")
        print(f"多风格回测: {start_date} 至 {end_date}")
        print(f"{'='*60}")

        for style_name in style_names:
            self.backtest_style(style_name, start_date, end_date)

        # 对比结果
        self._compare_results()

        return self.results

    def _calculate_performance(self, trades, final_capital, initial_capital):
        """计算绩效指标"""

        if not trades:
            return {
                'total_return': 0,
                'win_rate': 0,
                'avg_profit': 0,
                'max_drawdown': 0,
                'sharpe': 0,
                'trade_count': 0
            }

        # 收益率
        total_return = (final_capital - initial_capital) / initial_capital

        # 胜率
        wins = [t for t in trades if t['profit'] > 0]
        win_rate = len(wins) / len(trades)

        # 平均盈亏
        avg_profit = sum(t['profit'] for t in trades) / len(trades)

        # 最大回撤（简化）
        max_drawdown = min(t['profit'] for t in trades) if trades else 0

        # 夏普比率（简化）
        returns = [t['profit'] for t in trades]
        sharpe = (sum(returns) / len(returns)) / (max(returns) - min(returns) + 0.001) if returns else 0

        return {
            'total_return': total_return,
            'win_rate': win_rate,
            'avg_profit': avg_profit,
            'max_drawdown': max_drawdown,
            'sharpe': sharpe,
            'trade_count': len(trades)
        }

    def _print_result(self, result):
        """打印回测结果"""
        print(f"\n【{result['style']} 回测结果】")
        print(f"  总收益率: {result['total_return']:.2%}")
        print(f"  胜率: {result['win_rate']:.1%}")
        print(f"  平均盈亏: {result['avg_profit']:.2%}")
        print(f"  最大回撤: {result['max_drawdown']:.2%}")
        print(f"  夏普比率: {result['sharpe']:.2f}")
        print(f"  交易次数: {result['trade_count']}")

    def _compare_results(self):
        """对比多种风格的结果"""

        print(f"\n{'='*60}")
        print("多风格对比")
        print(f"{'='*60}")

        # 按收益率排序
        sorted_results = sorted(
            self.results.items(),
            key=lambda x: x[1]['total_return'],
            reverse=True
        )

        print(f"\n【收益率排名】")
        for i, (style, result) in enumerate(sorted_results, 1):
            print(f"  {i}. {style}: {result['total_return']:.2%}")

        # 按夏普排序
        sorted_sharpe = sorted(
            self.results.items(),
            key=lambda x: x[1]['sharpe'],
            reverse=True
        )

        print(f"\n【夏普比率排名】")
        for i, (style, result) in enumerate(sorted_sharpe, 1):
            print(f"  {i}. {style}: {result['sharpe']:.2f}")

        # 找出最佳风格
        best_return = sorted_results[0][0]
        best_sharpe = sorted_sharpe[0][0]

        print(f"\n【最佳风格】")
        print(f"  收益最佳: {best_return}")
        print(f"  稳定性最佳: {best_sharpe}")

        if best_return == best_sharpe:
            print(f"  综合最佳: {best_return}")
        else:
            print(f"  建议: 收益优先选 {best_return}, 稳健优先选 {best_sharpe}")

    """风格优化器 - 自动寻找最适合当前市场的风格"""

    def __init__(self, style_def, backtester):
        self.style_def = style_def
        self.backtester = backtester
        self.optimization_history = []

    def find_best_style(self, recent_data, lookback_days=60, optimization_criteria='sharpe'):
        """寻找最近一段时间的最佳风格"""

        print(f"\n===== 风格优化 =====")
        print(f"分析最近 {lookback_days} 天的市场数据")

        end_date = recent_data['end_date']
        start_date = end_date - timedelta(days=lookback_days)

        # 获取所有风格
        all_styles = self.style_def.list_styles()

        # 回测所有风格
        results = {}
        for style in all_styles:
            result = self.backtester.backtest_style(
                style, start_date, end_date,
                initial_capital=100000
            )
            if result:
                results[style] = result

        # 按指定标准排序
        if optimization_criteria == 'sharpe':
            sorted_styles = sorted(
                results.items(),
                key=lambda x: x[1]['sharpe'],
                reverse=True
            )
        elif optimization_criteria == 'return':
            sorted_styles = sorted(
                results.items(),
                key=lambda x: x[1]['total_return'],
                reverse=True
            )
        else:  # 综合评分
            sorted_styles = sorted(
                results.items(),
                key=lambda x: x[1]['sharpe'] * x[1]['total_return'],
                reverse=True
            )

        # 记录优化结果
        optimization = {
            'date': end_date,
            'lookback_days': lookback_days,
            'criteria': optimization_criteria,
            'ranking': [(s, r['sharpe'], r['total_return']) for s, r in sorted_styles],
            'best_style': sorted_styles[0][0] if sorted_styles else None
        }

        self.optimization_history.append(optimization)

        # 打印结果
        print(f"\n【最佳风格: {optimization['best_style']}】")
        print(f"\n【风格排名】")
        for i, (style, sharpe, ret) in enumerate(optimization['ranking'][:5], 1):
            print(f"  {i}. {style}: 夏普{sharpe:.2f} 收益{ret:.2%}")

        return optimization

    def adaptive_style_selection(self, current_data, window_sizes=[20, 60, 120]):
        """自适应风格选择 - 考虑多个时间窗口"""

        print(f"\n===== 自适应风格选择 =====")

        votes = {}

        for window in window_sizes:
            print(f"\n分析 {window} 天窗口")
            opt = self.find_best_style(current_data, window, 'sharpe')

            if opt['best_style']:
                votes[opt['best_style']] = votes.get(opt['best_style'], 0) + 1

        # 统计得票
        print(f"\n【风格投票结果】")
        sorted_votes = sorted(votes.items(), key=lambda x: x[1], reverse=True)

        for style, vote_count in sorted_votes:
            print(f"  {style}: {vote_count}票")

        if sorted_votes:
            recommended = sorted_votes[0][0]
            print(f"\n推荐风格: {recommended}")
            return recommended

        return None

    def style_rotation_signal(self, market_data):
        """风格轮动信号 - 何时切换风格"""

        # 检测市场变化
        market_regime = self._detect_market_regime(market_data)

        # 风格-市场映射
        style_market_map = {
            '上涨': ['长线', '中线', '成长投资'],
            '震荡': ['短线', '超短线'],
            '下跌': ['价值投资', '空仓'],
            '极端': ['超短线', '空仓']
        }

        # 获取适合当前市场的风格
        suitable_styles = style_market_map.get(market_regime, ['短线'])

        print(f"\n【市场状态: {market_regime}】")
        print(f"适合风格: {suitable_styles}")

        return {
            'market_regime': market_regime,
            'suitable_styles': suitable_styles,
            'should_rotate': True
        }

    def _detect_market_regime(self, data):
        """检测市场状态"""

        # 计算关键指标
        returns = data['index'].pct_change()
        volatility = returns.std() * 100

        trend = (data['index'][-1] / data['index'][-20] - 1) if len(data['index']) >= 20 else 0

        if volatility > 3:
            return '极端'
        elif trend > 0.05:
            return '上涨'
        elif trend < -0.05:
            return '下跌'
        else:
            return '震荡'

class StyleCombiner:
    """风格组合器 - 多风格资金分配和运行"""

    def __init__(self, style_def, style_adapter):
        self.style_def = style_def
        self.style_adapter = style_adapter

        self.portfolio = {}  # {style_name: weight}
        self.style_systems = {}  # {style_name: adapted_system}
        self.performance = {}  # {style_name: performance}

    def create_portfolio(self, style_weights):
        """创建投资组合，分配权重"""

        total_weight = sum(style_weights.values())
        if abs(total_weight - 1.0) > 0.01:
            print(f"警告: 权重总和为{total_weight}，将自动归一化")
            for style in style_weights:
                style_weights[style] /= total_weight

        self.portfolio = style_weights

        print(f"\n===== 多风格投资组合 =====")
        for style, weight in style_weights.items():
            print(f"  {style}: {weight:.1%}")

        return self.portfolio

    def allocate_capital(self, total_capital):
        """根据权重分配资金"""

        allocation = {}
        for style, weight in self.portfolio.items():
            allocation[style] = total_capital * weight

        print(f"\n【资金分配】")
        for style, capital in allocation.items():
            print(f"  {style}: {capital:,.0f}")

        return allocation

    def run_multi_style_system(self, market_data):
        """运行多风格系统"""

        decisions = []

        for style, weight in self.portfolio.items():
            # 获取该风格的适配系统
            if style not in self.style_systems:
                system = self._create_style_system(style)
                self.style_systems[style] = system

            system = self.style_systems[style]

            # 运行系统
            decision = system.on_tick(market_data['tick'], market_data)

            if decision:
                decision['style'] = style
                decision['weight'] = weight
                decisions.append(decision)

        # 综合决策
        final_decisions = self._integrate_decisions(decisions)

        return final_decisions

    def _create_style_system(self, style):
        """为特定风格创建系统"""
        self.style_adapter.set_style(style)

        system = {
            'style': style,
            'positions': [],
            'last_decision': None
        }

        return system

    def _integrate_decisions(self, decisions):
        """综合多个风格的决策"""

        # 按股票合并
        stock_decisions = {}

        for d in decisions:
            symbol = d.get('symbol')
            if not symbol:
                continue

            if symbol not in stock_decisions:
                stock_decisions[symbol] = []

            stock_decisions[symbol].append(d)

        # 对每只股票，综合决策
        final = []
        for symbol, sym_decisions in stock_decisions.items():
            # 加权投票
            buy_weight = sum(d['weight'] for d in sym_decisions if d['action'] == 'buy')
            sell_weight = sum(d['weight'] for d in sym_decisions if d['action'] == 'sell')
            hold_weight = sum(d['weight'] for d in sym_decisions if d['action'] == 'hold')

            if buy_weight > 0.5:
                final.append({
                    'symbol': symbol,
                    'action': 'buy',
                    'confidence': buy_weight,
                    'reasons': [d['reason'] for d in sym_decisions if d['action'] == 'buy']
                })
            elif sell_weight > 0.5:
                final.append({
                    'symbol': symbol,
                    'action': 'sell',
                    'confidence': sell_weight,
                    'reasons': [d['reason'] for d in sym_decisions if d['action'] == 'sell']
                })
            else:
                final.append({
                    'symbol': symbol,
                    'action': 'hold',
                    'confidence': max(buy_weight, sell_weight, hold_weight),
                    'reasons': ['多风格意见分歧']
                })

        return final

    def track_style_performance(self):
        """跟踪各风格的表现"""

        for style, system in self.style_systems.items():
            if hasattr(system, 'get_performance'):
                perf = system.get_performance()
                self.performance[style] = perf

        return self.performance

    def rebalance_portfolio(self, lookback_days=30):
        """根据近期表现重新平衡组合"""

        if not self.performance:
            return self.portfolio

        # 获取近期各风格的收益率
        recent_perf = {}
        for style, perf in self.performance.items():
            if style in self.portfolio:
                recent_perf[style] = perf.get('recent_return', 0)

        if not recent_perf:
            return self.portfolio

        # 根据表现调整权重
        total_perf = sum(recent_perf.values())
        if total_perf <= 0:
            return self.portfolio

        new_weights = {}
        for style, perf in recent_perf.items():
            new_weights[style] = perf / total_perf

        # 归一化
        total = sum(new_weights.values())
        for style in new_weights:
            new_weights[style] /= total

        print(f"\n===== 组合再平衡 =====")
        print(f"原权重: {self.portfolio}")
        print(f"新权重: {new_weights}")

        self.portfolio = new_weights
        return new_weights

class SurvivalGuarantee:
    """生存保障系统 - 绝对不能输光"""

    def __init__(self, initial_capital):
        self.initial_capital = initial_capital
        self.current_capital = initial_capital
        self.peak_capital = initial_capital

        # 生存红线
        self.survival_lines = {
            'red_line': initial_capital * 0.7,      # 70% - 警告线
            'yellow_line': initial_capital * 0.5,    # 50% - 危险线
            'black_line': initial_capital * 0.3,     # 30% - 生死线
            'game_over': initial_capital * 0.1       # 10% - 出局
        }

        # 生存状态
        self.status = {
            'level': '安全',
            'warnings': [],
            'can_trade': True,
            'reason': ''
        }

    def update(self, current_capital):
        """更新生存状态，检查生存线"""

        # 检查各条红线
        if current_capital <= self.survival_lines['game_over']:
            self._trigger_emergency_stop('game_over')
        elif current_capital <= self.survival_lines['black_line']:
            self._trigger_emergency_stop('black_line')
        elif current_capital <= self.survival_lines['yellow_line']:
            self._enter_danger_zone()
        elif current_capital <= self.survival_lines['red_line']:
            self._enter_warning_zone()

    def check_before_trade(self, trade_amount):
        """交易前检查"""
        # 检查交易是否会导致跌破生存线
        new_capital = self.current_capital - trade_amount * 0.03

        if new_capital <= self.survival_lines['black_line']:
            return {'allowed': False, 'reason': '交易可能导致进入生死线'}

        return {'allowed': True}

    def get_safe_position_size(self, confidence):
        """获取安全仓位大小"""
        base_size = 0.3

            base_size = 0.1  # 10%

            base_size = 0.2  # 20%

### 23.8 熔断机制

```python
class CircuitBreaker:
    """熔断机制 - 防止连续亏损"""

    def __init__(self):
        self.daily_loss_limit = 0.05      # 单日亏损5%熔断
        self.weekly_loss_limit = 0.1       # 单周亏损10%熔断
        self.monthly_loss_limit = 0.15     # 单月亏损15%熔断
        self.consecutive_loss_limit = 3     # 连续3次亏损熔断

    def on_trade(self, trade_result):
        """每笔交易后检查"""

        if trade_result['profit'] < 0:
            self.consecutive_losses += 1
            self.daily_loss += abs(trade_result['profit'])
        else:
            self.consecutive_losses = 0

        # 检查熔断条件
        if self.consecutive_losses >= self.consecutive_loss_limit:
            self._trigger_breaker('consecutive')

        if self.daily_loss >= self.daily_loss_limit:
            self._trigger_breaker('daily')

    def _trigger_breaker(self, breaker_type, reason):
        """触发熔断"""
        print(f"🔴 熔断触发 [{breaker_type}]: {reason}")
        print("操作: 立即停止所有交易，等待下一周期")

    def can_trade(self):
        """是否可以交易"""
        return not any([self.daily_break, self.weekly_break, self.monthly_break])
```

                'volatility': self._analyze_volatility(market_state), # 波动基因
                'liquidity': self._analyze_liquidity(market_state),   # 流动性基因
                'sentiment': self._analyze_sentiment(market_state),   # 情绪基因
                'structure': self._analyze_structure(market_state),   # 结构基因
                'cycle': self._analyze_cycle(market_state)            # 周期基因
            },
            'fingerprint': '',  # 市场指纹
            'regime': ''        # 市场状态
        }

        # 生成市场指纹
        market_dna['fingerprint'] = self._generate_fingerprint(
            market_dna['components']
        )

        # 识别市场状态
        market_dna['regime'] = self._identify_regime(
            market_dna['components']
        )

        return market_dna

    def _analyze_trend(self, data):
        """分析趋势基因 - 方向、强度、持续性"""
        prices = data.get('price_history', [])
        short_ma = sum(prices[-5:]) / 5 if len(prices) >= 5 else prices[-1]
        long_ma = sum(prices[-20:]) / 20 if len(prices) >= 20 else prices[-1]

        strength = abs(short_ma - long_ma) / long_ma
        direction = 'up' if short_ma > long_ma else 'down'

        return {
            'strength': min(strength * 10, 1),
            'direction': direction,
            'consistency': self._calc_trend_consistency(prices),
            'phase': self._get_trend_phase(prices)
        }

    def _analyze_volatility(self, data):
        """分析波动基因 - 水平、结构、聚集"""
        returns = data.get('returns', [])
        volatility = np.std(returns) * np.sqrt(252) if returns else 0.15

        if volatility < 0.15:
            level = 'low'
        elif volatility < 0.25:
            level = 'normal'
        elif volatility < 0.35:
            level = 'high'
        else:
            level = 'extreme'

        return {
            'value': volatility,
            'level': level,
            'structure': self._analyze_volatility_structure(returns),
            'cluster': self._detect_volatility_cluster(returns)
        }

    def _generate_fingerprint(self, components):
        """生成市场指纹 - 唯一标识当前市场状态"""
        trend = components['trend']
        vol = components['volatility']
        liq = components['liquidity']
        sent = components['sentiment']
        struct = components['structure']

        return f"{trend['direction'][:1]}{trend['strength']:.1f}_{vol['level'][:1]}{vol['value']:.2f}_{liq['level'][:2]}{liq['ratio']:.1f}_{sent['state'][:2]}{sent['score']:.0f}_{struct['type'][:2]}"

    def _identify_regime(self, components):
        """识别市场状态"""
        trend_dir = components['trend']['direction']
        vol_level = components['volatility']['level']

        if trend_dir == 'up':
            if vol_level == 'low':
                return 'steady_bull'
            elif vol_level == 'normal':
                return 'normal_bull'
            else:
                return 'volatile_bull'
        elif trend_dir == 'down':
            if vol_level == 'low':
                return 'steady_bear'
            elif vol_level == 'normal':
                return 'normal_bear'
            else:
                return 'panic_bear'
        else:
            if vol_level == 'low':
                return 'quiet_range'
            else:
                return 'chaotic_range'

        if turnover_std < 0.5:
            score += 0.2

        big_buy_ratio = data['big_buy_volume'].sum() / data['volume'].sum()
        if big_buy_ratio > 0.2:
            score += 0.2

class InstitutionCostCalculator:
    """机构成本测算器"""

    def calculate(self, stock_data):
        """计算机构成本区"""

        methods = []

        # 方法1: 用量价关系算成本
        volume_cost = self._calc_by_volume_price(stock_data)
        if volume_cost:
            methods.append(volume_cost)

        # 方法2: 用时间加权算成本
        time_cost = self._calc_by_time_weight(stock_data)
        if time_cost:
            methods.append(time_cost)

        # 方法3: 用回调位置算成本
        retrace_cost = self._calc_by_retracement(stock_data)
        if retrace_cost:
            methods.append(retrace_cost)

        if methods:
            avg_cost = sum(methods) / len(methods)
            return {
                'lower': avg_cost * 0.95,
                'center': avg_cost,
                'upper': avg_cost * 1.05
            }

        return None

    def _calc_by_volume_price(self, data):
        """用量价关系算成本"""
        volume = data['volume']
        price = data['close']
        volume_ma = volume.rolling(20).mean()
        volume_spike = volume > volume_ma * 1.5

        if any(volume_spike):
            spike_prices = price[volume_spike]
            spike_volumes = volume[volume_spike]
            return sum(spike_prices * spike_volumes) / sum(spike_volumes)

        return None

    def _calc_by_time_weight(self, data):
        """用时间加权算成本"""
        recent = data.iloc[-60:]
        weights = range(1, len(recent) + 1)
        weights = [w / sum(weights) for w in weights]
        return sum(recent['close'] * weights)

    def _calc_by_retracement(self, data):
        """用回调位置算成本"""
        high = data['high'].max()
        low = data['low'].min()
        return low + (high - low) * 0.382  # 黄金分割位

    def get_advice(self, stock_data):
        """获取建议"""
        cost_info = self.calculate(stock_data)
        current_price = stock_data['close'].iloc[-1]

        if not cost_info:
            return "无法计算机构成本"

            return "💰 股价低于机构成本区，可以大胆买入"
        elif current_price < cost_info['center']:
            return "💎 股价在机构成本区下沿，可以分批买入"

            return "📊 股价在机构成本区内，可以持有"

class InstitutionStockFilter:
    """机构票四维筛选器 - 把大海变成鱼缸"""

    def __init__(self):
        self.dimensions = {
            '机构质量': self._filter_quality,
            '买入力度': self._filter_strength,
            '位置安全': self._filter_position,
            '后续表现': self._filter_performance
        }

    def filter(self, stock_list):
        """筛选机构票"""

        candidates = []

        for stock in stock_list:
            score = 0
            reasons = []

            # 1. 机构质量
            quality = self._filter_quality(stock)
            score += quality['score']
            if quality['score'] > 0:
                reasons.append(quality['reason'])

            # 2. 买入力度
            strength = self._filter_strength(stock)
            score += strength['score']
            if strength['score'] > 0:
                reasons.append(strength['reason'])

            # 3. 位置安全
            position = self._filter_position(stock)
            score += position['score']
            if position['score'] > 0:
                reasons.append(position['reason'])

            # 4. 后续表现
            performance = self._filter_performance(stock)
            score += performance['score']
            if performance['score'] > 0:
                reasons.append(performance['reason'])

            if score >= 6:  # 总分8分，6分以上入选
                candidates.append({
                    'stock': stock['code'],
                    'score': score,
                    'reasons': reasons,
                    'level': 'A类' if score >= 7 else 'B类'
                })

        return sorted(candidates, key=lambda x: x['score'], reverse=True)

    def _filter_quality(self, stock):
        """过滤机构质量"""
        score = 0
        reason = ""

        buyers = stock.get('institution_buyers', [])

        # 顶级机构买入
        top_count = sum(1 for b in buyers if b['level'] == '顶级机构')
        if top_count >= 2:
            score += 3
            reason = f"{top_count}家顶级机构买入"
        elif top_count >= 1:
            score += 2
            reason = "有顶级机构买入"

        # 主力机构买入
        main_count = sum(1 for b in buyers if b['level'] == '主力机构')
        if main_count >= 3:
            score += 2
            reason += "，多家主力机构跟进"
        elif main_count >= 1:
            score += 1
            reason += "，有主力机构参与"

        # 没有游资和假机构
        youzi_count = sum(1 for b in buyers if b['level'] == '游资机构')
        fake_count = sum(1 for b in buyers if b['level'] == '普通席位')

        if youzi_count == 0 and fake_count == 0:
            score += 1
            reason += "，无游资杂毛"

        return {'score': score, 'reason': reason}

    def _filter_strength(self, stock):
        """过滤买入力度"""
        score = 0
        reason = ""

        buy_amount = stock.get('total_buy', 0)
        buy_ratio = stock.get('buy_ratio', 0)

        # 买入金额
        if buy_amount > 2e8:  # 2亿以上
            score += 3
            reason = f"机构买入{buy_amount/1e8:.1f}亿"
        elif buy_amount > 1e8:
            score += 2
            reason = f"机构买入{buy_amount/1e8:.1f}亿"
        elif buy_amount > 5e7:
            score += 1
            reason = f"机构买入{buy_amount/1e7:.0f}千万"

        # 买入占比
        if buy_ratio > 20:
            score += 2
            reason += f"，占比{buy_ratio}%"
        elif buy_ratio > 10:
            score += 1
            reason += f"，占比{buy_ratio}%"

        return {'score': score, 'reason': reason}

    def _filter_position(self, stock):
        """过滤位置安全"""
        score = 0
        reason = ""

        current_price = stock.get('price', 0)
        low_52w = stock.get('low_52w', 0)
        high_52w = stock.get('high_52w', 0)

        position_ratio = (current_price - low_52w) / (high_52w - low_52w) if high_52w > low_52w else 0.5

        if position_ratio < 0.3:
            score += 3
            reason = "历史低位"
        elif position_ratio < 0.5:
            score += 2
            reason = "相对低位"
        elif position_ratio > 0.8:
            score -= 2
            reason = "⚠️历史高位"

        return {'score': score, 'reason': reason}

    def _filter_performance(self, stock):
        """过滤后续表现"""
        score = 0
        reason = ""

        # 后续3天表现
        next_3d = stock.get('next_3d_return', 0)

        if next_3d > 0:
            score += 1
            reason = f"后续3天涨{next_3d:.1%}"
        elif next_3d < -0.05:
            score -= 1
            reason = f"后续3天跌{next_3d:.1%}"

        # 历史胜率
        win_rate = stock.get('institution_win_rate', 0.5)
        if win_rate > 0.7:
            score += 2
            reason += "，历史胜率高"

        return {'score': score, 'reason': reason}

                print(f"⚖️ 判决: {verdict}")

            return result

        else:
            print("⏳ 无交易信号，系统休息")
            return None

    def daily_report(self):
        """每日报告"""

        print("\n📊 系统日报")
        print("=" * 50)
        self.discipline.report()
        print("=" * 50)
