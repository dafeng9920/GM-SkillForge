                    job_id=job_id,
                    status="failed",
                    path_label="",
                    stages_completed=0,
                    total_stages=0,
                    duration_ms=int((time.time() - start_time) * 1000),

                    trace_events=trace_events,
                    gate_decisions=gate_decisions,

                    path_label="",

            total_stages = len(node_ids)

            # ── 5. init artifacts ─────────────────────────────────
            artifacts = {"input": input_data}

            for node_id in node_ids:
                stage_start = time.time()

                # 6a. check registration
                if node_id not in self.node_registry:
                    return self._build_output(

                        status="failed",
                        path_label=path_label,
                        stages_completed=stages_completed,

                        duration_ms=int((time.time() - start_time) * 1000),

                            trace_events=trace_events,
                            gate_decisions=gate_decisions,

                    if decision == "REQUIRES_CHANGES":

                            status="failed",

                        )

                # 6c. regular node (or gate without evaluator)
                else:
                    input_snapshot = dict(artifacts)

                    # validate input
                    try:
                        validation = handler.validate_input(input_snapshot)
                    except Exception as val_err:
                        return self._build_output(
                            job_id=job_id,
                            status="failed",
                            path_label=path_label,
                            stages_completed=stages_completed,
                            total_stages=total_stages,

                            artifacts=artifacts,
                            trace_events=trace_events,
                            gate_decisions=gate_decisions,
                            error=f"EXEC_VALIDATION_FAILED: {node_id}: {val_err}",
                        )

                    # normalise validation result
                    _failed = False
                    _detail = ""
                    if isinstance(validation, dict):
                        if not validation.get("valid", True):
                            _failed = True
                            _detail = str(validation.get("errors", []))
                    elif isinstance(validation, (list, tuple)):
                        if validation:
                            _failed = True
                            _detail = str(list(validation))

                    if _failed:
                        return self._build_output(
                            job_id=job_id,
                            status="failed",
                            path_label=path_label,
                            stages_completed=stages_completed,
                            total_stages=total_stages,
                            duration_ms=int(
                                (time.time() - start_time) * 1000
                            ),
                            artifacts=artifacts,
                            trace_events=trace_events,
                            gate_decisions=gate_decisions,
                            error=f"EXEC_VALIDATION_FAILED: {node_id}: {_detail}",
                        )

                    # execute
                    result = handler.execute(artifacts)

                    # validate output
                    try:
                        out_val = handler.validate_output(result)
                    except Exception as ov_err:
                        return self._build_output(
                            job_id=job_id,
                            status="failed",
                            path_label=path_label,
                            stages_completed=stages_completed,
                            total_stages=total_stages,
                            duration_ms=int(
                                (time.time() - start_time) * 1000
                            ),
                            artifacts=artifacts,
                            trace_events=trace_events,
                            gate_decisions=gate_decisions,
                            error=f"EXEC_OUTPUT_INVALID: {node_id}: {ov_err}",
                        )

                    _out_failed = False
                    if isinstance(out_val, dict) and not out_val.get(
                        "valid", True
                    ):
                        _out_failed = True
                        _detail = str(out_val.get("errors", []))
                    elif isinstance(out_val, (list, tuple)) and out_val:
                        _out_failed = True
                        _detail = str(list(out_val))

                    if _out_failed:
                        return self._build_output(
                            job_id=job_id,
                            status="failed",
                            path_label=path_label,
                            stages_completed=stages_completed,
                            total_stages=total_stages,
                            duration_ms=int(
                                (time.time() - start_time) * 1000
                            ),
                            artifacts=artifacts,
                            trace_events=trace_events,
                            gate_decisions=gate_decisions,
                            error=f"EXEC_OUTPUT_INVALID: {node_id}: {_detail}",
                        )

                    # store result
                    artifacts[node_id] = result

                # 6d. stages_completed
                stages_completed += 1

                # 6e. trace event
                stage_duration_ms = int(
                    (time.time() - stage_start) * 1000
                )
                trace_events.append(
                    {
                        "event_id": str(uuid.uuid4()),
                        "event_type": "complete",
                        "node_id": node_id,
                        "timestamp": time.time(),
                        "duration_ms": stage_duration_ms,
                    }
                )

            # ── 7. extract final outputs ──────────────────────────
            audit_pack_path: Optional[str] = None
            publish_result: Optional[dict[str, Any]] = None
            final_key = "pack_audit_and_publish"
            if final_key in artifacts and isinstance(
                artifacts[final_key], dict
            ):
                audit_pack_path = artifacts[final_key].get(
                    "audit_pack_path"
                )
                publish_result = artifacts[final_key].get("publish_result")

            # ── 8-9. build output ─────────────────────────────────
            duration_ms = int((time.time() - start_time) * 1000)
            return self._build_output(
                job_id=job_id,
                status="completed",
                path_label=path_label,

from __future__ import annotations

import time
import uuid
from datetime import datetime, timezone
from typing import Any, Optional


class NodeRunner:
    """Runs a single pipeline node with validation, retry, and tracing."""

    def __init__(
        self,
        default_timeout_seconds: int = 30,
        max_retries: int = 3,
        retry_delay_seconds: float = 1.0,
    ) -> None:
        """Initialize the node runner.

        Args:
            default_timeout_seconds: Default per-attempt timeout.
            max_retries: Default maximum number of execution attempts.
            retry_delay_seconds: Base delay between retries (exponential).
        """
        self.default_timeout_seconds = default_timeout_seconds
        self.max_retries = max_retries
        self.retry_delay_seconds = retry_delay_seconds

    # ── already implemented — do not modify ───────────────────────

    def _make_trace_event(
        self,
        *,
        event_type: str,
        node_id: str,
        duration_ms: Optional[int] = None,
        data: Optional[dict[str, Any]] = None,
        error: Optional[str] = None,
    ) -> dict[str, Any]:
        """Create a trace event dict conforming to trace_event.schema.json.

        Args:
            event_type: One of start, tool_call, llm_call, error,
                        complete, checkpoint.
            node_id: Identifier of the node that produced the event.
            duration_ms: Optional elapsed time in milliseconds.
            data: Optional payload dict.
            error: Optional error description.

        Returns:
            Trace event dictionary.
        """
        event: dict[str, Any] = {
            "event_id": str(uuid.uuid4()),
            "event_type": event_type,
            "node_id": node_id,
            "timestamp": datetime.now(timezone.utc).isoformat(),
        }
        if duration_ms is not None:
            event["duration_ms"] = duration_ms
        if data is not None:
            event["data"] = data
        if error is not None:
            event["error"] = error
        return event

    def run(
        self,
        handler: Any,
        input_data: dict[str, Any],
        *,
        timeout: Optional[int] = None,
        max_retries: Optional[int] = None,
    ) -> dict[str, Any]:
        """Execute a single node: validate → run (with retries) → validate output.

        Args:
            handler: Node handler instance.  Must expose ``node_id``,
                     ``validate_input``, ``execute``, and
                     ``validate_output``.
            input_data: Data dict to pass into the handler.
            timeout: Per-attempt timeout in seconds (falls back to
                     ``default_timeout_seconds``).
            max_retries: Maximum number of execution attempts (falls
                         back to ``self.max_retries``).

        Returns:
            Dict with keys ``node_id``, ``output``, ``trace_events``,
            ``duration_ms``, and ``retries_used``.

        Raises:
            ValueError: If input validation fails.
            RuntimeError: If execution fails after all retries or if
                          output validation fails.
        """
        # ── 1. resolve defaults ───────────────────────────────────
        effective_timeout = (
            timeout if timeout is not None else self.default_timeout_seconds
        )
        effective_max_retries = (
            max_retries if max_retries is not None else self.max_retries
        )

        node_id: str = getattr(handler, "node_id", type(handler).__name__)
        trace_events: list[dict[str, Any]] = []
        run_start = time.time()

        # ── 2. start trace ────────────────────────────────────────
        trace_events.append(
            self._make_trace_event(event_type="start", node_id=node_id)
        )

        # ── 3. validate input ─────────────────────────────────────
        try:
            validation = handler.validate_input(input_data)
        except Exception as val_exc:
            trace_events.append(
                self._make_trace_event(
                    event_type="error",
                    node_id=node_id,
                    error=f"EXEC_VALIDATION_FAILED: {val_exc}",
                )
            )
            raise ValueError(
                f"EXEC_VALIDATION_FAILED: {node_id}: {val_exc}"
            ) from val_exc

        # normalise: list-of-errors or dict with "valid"/"errors"
        _errors: list[str] = []
        if isinstance(validation, dict) and not validation.get("valid", True):
            _errors = validation.get("errors", ["input validation failed"])
        elif isinstance(validation, (list, tuple)) and validation:
            _errors = [str(e) for e in validation]

        if _errors:
            trace_events.append(
                self._make_trace_event(
                    event_type="error",
                    node_id=node_id,
                    error=f"EXEC_VALIDATION_FAILED: {_errors}",
                )
            )
            raise ValueError(
                f"EXEC_VALIDATION_FAILED: {node_id}: {_errors}"
            )

        # ── 4. retry loop ─────────────────────────────────────────
        retries_used = 0
        output: Any = None
        last_exc: Optional[Exception] = None

        for attempt in range(effective_max_retries):
            attempt_start = time.time()
            try:
                output = handler.execute(input_data)
                last_exc = None
                break
            except Exception as exec_exc:
                last_exc = exec_exc
                retries_used += 1

                # check simplified timeout
                elapsed = time.time() - attempt_start
                if elapsed > effective_timeout:
                    trace_events.append(
                        self._make_trace_event(
                            event_type="error",
                            node_id=node_id,
                            error=f"EXEC_TIMEOUT: attempt {attempt + 1} "
                            f"exceeded {effective_timeout}s",
                        )
                    )

                if attempt < effective_max_retries - 1:
                    # exponential back-off
                    delay = self.retry_delay_seconds * (2 ** retries_used)
                    time.sleep(delay)

        if last_exc is not None:
            trace_events.append(
                self._make_trace_event(
                    event_type="error",
                    node_id=node_id,
                    error=f"EXEC_RETRIES_EXHAUSTED: {last_exc}",
                )
            )
            raise RuntimeError(
                f"EXEC_RETRIES_EXHAUSTED: {node_id}: "
                f"failed after {retries_used} attempts: {last_exc}"
            ) from last_exc

        # ── 5. validate output ────────────────────────────────────
        try:
            out_validation = handler.validate_output(output)
        except Exception as ov_exc:
            trace_events.append(
                self._make_trace_event(
                    event_type="error",
                    node_id=node_id,
                    error=f"EXEC_OUTPUT_INVALID: {ov_exc}",
                )
            )
            raise RuntimeError(
                f"EXEC_OUTPUT_INVALID: {node_id}: {ov_exc}"
            ) from ov_exc

        _out_errors: list[str] = []
        if isinstance(out_validation, dict) and not out_validation.get(
            "valid", True
        ):
            _out_errors = out_validation.get(
                "errors", ["output validation failed"]
            )
        elif isinstance(out_validation, (list, tuple)) and out_validation:
            _out_errors = [str(e) for e in out_validation]

        if _out_errors:
            trace_events.append(
                self._make_trace_event(
                    event_type="error",
                    node_id=node_id,
                    error=f"EXEC_OUTPUT_INVALID: {_out_errors}",
                )
            )
            raise RuntimeError(
                f"EXEC_OUTPUT_INVALID: {node_id}: {_out_errors}"
            )

        # ── 6. complete trace ─────────────────────────────────────
        total_duration_ms = int((time.time() - run_start) * 1000)
        trace_events.append(
            self._make_trace_event(
                event_type="complete",
                node_id=node_id,
                duration_ms=total_duration_ms,
            )
        )

        # ── 7. return ─────────────────────────────────────────────
        return {
            "node_id": node_id,
            "output": output,
            "trace_events": trace_events,
            "duration_ms": total_duration_ms,
            "retries_used": retries_used,
        }
